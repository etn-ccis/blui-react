1    	import React, { useCallback, useState } from 'react';
2    	import { EulaScreenProps } from './types';
3    	import {
4    	    WorkflowCard,
5    	    WorkflowCardActions,
6    	    WorkflowCardBody,
7    	    WorkflowCardHeader,
8    	    WorkflowCardInstructions,
9    	} from '../../components/WorkflowCard';
10   	import FormControlLabel from '@mui/material/FormControlLabel';
11   	import Checkbox from '@mui/material/Checkbox';
12   	import Box from '@mui/material/Box';
13   	import DOMPurify from 'dompurify';
14   	import ErrorManager from '../../components/Error/ErrorManager';
15   	import { Typography } from '@mui/material';
16   	import ReplaySharp from '@mui/icons-material/ReplaySharp';
17   	import { useTranslation } from 'react-i18next';
18   	
19   	/**
20   	 * Component that renders a screen displaying the EULA and requests acceptance via a checkbox.
21   	 *
22   	 * @param {EulaScreenProps} props - props of EUlA screen base component
23   	 *
24   	 * @category Component
25   	 */
26   	
27   	export const EulaScreenBase: React.FC<EulaScreenProps> = (props) => {
28   	    const {
29   	        onEulaAcceptedChange,
30   	        eulaContent,
31   	        checkboxLabel,
32   	        html,
33   	        initialCheckboxValue,
34   	        checkboxProps,
35   	        errorDisplayConfig,
36   	        refreshConfig,
37   	        WorkflowCardBaseProps: cardBaseProps = {},
38   	        WorkflowCardInstructionProps: instructionsProps = {},
39   	        WorkflowCardActionsProps: actionsProps = {},
40   	        WorkflowCardHeaderProps: headerProps = {},
41   	        ...otherProps
42   	    } = props;
43   	
44   	    const { t } = useTranslation();
45   	
46   	    const [eulaAccepted, setEulaAccepted] = useState(initialCheckboxValue ?? false);
47   	
48   	    const handleEulaAcceptedChecked = useCallback(
49   	        (accepted: boolean) => {
50   	            setEulaAccepted(accepted);
51   	            onEulaAcceptedChange?.(accepted);
52   	        },
53   	        [onEulaAcceptedChange]
54   	    );
55   	
56   	    const handleOnNext = (): void => {
57   	        const { onNext } = actionsProps;
58   	        if (onNext) onNext({ accepted: eulaAccepted });
59   	    };
60   	
61   	    return (
62   	        <WorkflowCard {...cardBaseProps} {...otherProps}>
63   	            <WorkflowCardHeader {...headerProps} />
64   	            {Object.keys(instructionsProps).length !== 0 && <WorkflowCardInstructions {...instructionsProps} />}
65   	            <WorkflowCardBody sx={{ pt: 2 }}>
66   	                {refreshConfig?.showRefreshButton ? (
67   	                    <Box
68   	                        sx={{
69   	                            display: 'flex',
70   	                            flexDirection: 'column',
71   	                            alignItems: 'center',
72   	                            justifyContent: 'center',
73   	                            height: '100%',
74   	                        }}
75   	                    >
76   	                        <Box
77   	                            sx={{
78   	                                flexDirection: 'column',
79   	                                alignItems: 'center',
80   	                                justifyContent: 'center',
81   	                                cursor: 'pointer',
82   	                                p: 1,
83   	                            }}
84   	                            onClick={refreshConfig?.onRefresh}
85   	                        >
86   	                            <ReplaySharp color="primary" sx={{ width: 36, height: 36 }} />
87   	                            <Typography variant="subtitle2" color="primary">
88   	                                {refreshConfig?.refreshButtonLabel || t('bluiCommon:MESSAGES.RETRY')}
89   	                            </Typography>
90   	                        </Box>
91   	                    </Box>
92   	                ) : html ? (
93   	                    <Box
94   	                        sx={{ flex: '1 1 0', overflow: 'auto' }}
95   	                        // eslint-disable-next-line @typescript-eslint/naming-convention
(1) Event Sigma main event: Setting raw HTML content without proper sanitization to the `innerHTML` property of a DOM element by using the `dangerouslySetInnerHTML` attribute may lead to cross-site scripting vulnerabilities.
(2) Event remediation: Audit the usage of `dangerouslySetInnerHTML` to make sure it is necessary and its content is properly sanitized. 
96   	                        dangerouslySetInnerHTML={{ __html: DOMPurify.sanitize(eulaContent as string) }}
97   	                    />
98   	                ) : (
99   	                    <Box sx={{ flex: '1 1 0', overflow: 'auto' }}>{eulaContent}</Box>
100  	                )}
101  	                <ErrorManager {...errorDisplayConfig}>
102  	                    <FormControlLabel
103  	                        control={
104  	                            <Checkbox
105  	                                color={'primary'}
106  	                                checked={eulaAccepted}
107  	                                onChange={(event: React.ChangeEvent<HTMLInputElement>): void =>
108  	                                    handleEulaAcceptedChecked(event.target.checked)
109  	                                }
110  	                                {...checkboxProps}
111  	                            />
112  	                        }
113  	                        label={checkboxLabel}
114  	                        sx={{ flex: '0 0 auto', mr: 0, mt: 2 }}
115  	                        onKeyUp={(e): void => {
116  	                            if (e.key === 'Enter' && eulaAccepted) handleOnNext();
117  	                        }}
118  	                    />
119  	                </ErrorManager>
120  	            </WorkflowCardBody>
121  	            <WorkflowCardActions {...actionsProps} canGoNext={eulaAccepted && actionsProps.canGoNext} />
122  	        </WorkflowCard>
123  	    );
124  	};
