62   	    };

133  	
134  	    return (

146  	            }}
147  	            enableButton={checkPasswords}
148  	            onPrevious={onPrevious}
149  	            PasswordProps={passwordProps}

161  	                    },
162  	                    onDismiss: (): void => {

164  	                    },
165  	                    WorkflowCardActionsProps: {
166  	                        showPrevious: false,
167  	                        fullWidthButton: true,
168  	                        showNext: true,
169  	                        nextLabel: t('bluiCommon:ACTIONS.DONE'),
170  	                        onNext: (): void => {
171  	                            onFinish?.();
172  	                        },
173  	                    },
174  	                    ...slotProps.SuccessScreen,
175  	                },
176  	            }}
177  	            showSuccessScreen={showSuccessScreen}
178  	            errorDisplayConfig={errorDisplayConfig}
