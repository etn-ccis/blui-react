9    	/**

120  	            initialValue={
121  	                screenData.CreateAccount.emailAddress.length > 0 ? screenData.CreateAccount.emailAddress : initialValue
122  	            }
123  	            emailTextFieldProps={{ ...emailTextFieldProps, onChange: onEmailInputValueChange }}
124  	            emailValidator={emailValidator}
125  	            WorkflowCardActionsProps={workflowCardActionsProps}
126  	            {...otherProps}
127  	            errorDisplayConfig={errorDisplayConfig}
128  	        />
